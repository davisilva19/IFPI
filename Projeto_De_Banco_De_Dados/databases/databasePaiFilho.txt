create database pai_filho;

use pai_filho;

create table pai (
  id_pai int not null,
  pai varchar(35),
  salario numeric(9,2),
  primary key(id_pai)
);

create table filho (
 Id_filho int not null,
 id_pai int,
 filho varchar(35),
 idade int,
 primary key(id_filho),
 foreign key(id_pai) references pai(Id_pai)
);

select * from pai;
